CREATE SEQUENCE carrito_seq
START WITH 1
INCREMENT BY 1;

CREATE OR REPLACE TRIGGER carrito_trg
BEFORE INSERT ON carrito
FOR EACH ROW
DECLARE
  next_id INTEGER;
BEGIN
  SELECT carrito_seq.NEXTVAL
  INTO next_id
  FROM dual
  FOR UPDATE;

  :new.id := next_id;
END;
