CREATE SEQUENCE libro_seq
START WITH 1
INCREMENT BY 1;


CREATE OR REPLACE TRIGGER libro_trg
BEFORE INSERT ON libro
FOR EACH ROW
BEGIN
    SELECT libro_seq.NEXTVAL
    INTO :new.id
    FROM dual;
END;
